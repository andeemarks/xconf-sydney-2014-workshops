Prerequisites

  • Laptop with Vagrant/Virtual Box installed
  • Access to smb://sydfs02/shared/TWA Misc/XCONF 2014

Create developer box


# Copy Vagrant stuff for dev box
cd $XCONF_HOME
cp /Volumes/XCONF\ 2014/xconf-dev.box .
cp /Volumes/XCONF\ 2014/Vagrantfile .
						

VAGRANTFILE_API_VERSION = "2"

Vagrant.configure(VAGRANTFILE_API_VERSION) do |config|
  config.vm.box = "centos-go-1.2.2"
  config.vm.box_url = "xconf-dev.box"
  config.vm.provider "virtualbox" do |vb|
    vb.customize ["modifyvm", :id, "--memory", "512"]
  end
  config.vm.network "forwarded_port", guest: 80, host: 8080
  config.vm.network "forwarded_port", guest: 8080, host: 8081
end
						

# Start Vagrant
vagrant up && vagrant ssh

Back-end service


# Clone repo
mkdir -p $GOPATH/src/github.com/andeemarks
cd $GOPATH/src/github.com/andeemarks
git clone https://github.com/andeemarks/xconf-go-svc.git
						

# Build service
cd xconf-go-svc
godep restore
go install
						

# Run automated tests
go test

Back-end service


# Start service
cd $GOPATH/bin
./xconf-go-svc &
