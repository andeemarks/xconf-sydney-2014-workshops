Install developer box


# Note: XCONF_HOME is a working folder on your host laptop
cd $XCONF_HOME 
# Check pre-reqs
vagrant version
# => Installed Version: 1.6.3
VBoxManage -v
# => 4.3.14r95030
# Have we got the files?
ls 
# => Vagrantfile     dev-sandbox.box
						

# Start Vagrant
vagrant up && vagrant ssh
